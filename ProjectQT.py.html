18   # This function loads the ECG data files: .dat, and .hea -files.
19   # .dat files consists of the ECG voltage data
20   # .hea files consist of the header information (sampling frequency, leads etc.)
21   def loadEcg(path):
22       dat_files = glob.glob(os.path.join(path, "*.dat"))
23       hea_files = glob.glob(os.path.join(path, "*.hea"))
24       base_names = set([os.path.splitext(os.path.basename(f))[0] for f in dat_files + hea_files])
25   
26       ecg_dict, field_dict, fs_dict = {}, {}, {}
27   
28       # Read the signal and metadata for each file. The read file consist of field names (contain, e.g.,
29       # the sampling frequency and the lead names), and the actual ecg signal data.
30       for i, base_name in enumerate(sorted(base_names), start=1):
31           ecg, fields = wfdb.rdsamp(os.path.join(path, base_name))
32           patient_key = f'Patient{i}'
33           ecg_dict[patient_key] = ecg
34           field_dict[patient_key] = fields
35           fs_dict[patient_key] = fields['fs']
36   
37       return ecg_dict, field_dict, fs_dict

40   # Function to filter ECG signal with FIR filter
41   # Change the cutoffs: low = 0.5 and high = 150 (bandpass FIR filtering)
42   # cutoffs are selected due to report instructions
43   def filterEcg(key, signal, fs, filter_order=20, low=0.5, high= 150):
44       # build a filter
45       filter_coeffs = firwin(filter_order, [low, high], fs=fs, pass_zero=False)
46   
47       # compare the filtered and the original ecg signals by plotting them
48       # Latter: Use Lead II (index 1) and apply the filter.
49       leadII = signal[:,1]
50       filtered_signal = lfilter(filter_coeffs, [1.0], leadII)
51   
52       time = np.arange(leadII.size) / fs  # creating a time vector
53   
54       # Plotting and comparing the filtering process among specific patients
55       if key in ["Patient1", "Patient2", "Patient3"]:
56           plt.figure(figsize=(10, 6))
57           plt.plot(time, leadII)
58           plt.title(f"{key}: Original ECG Lead II")
59           plt.xlabel('Time (s)')
60           plt.ylabel('Amplitude')
61           plt.show()
62   
63           plt.figure(figsize=(10, 6))
64           plt.plot(time, filtered_signal)
65           plt.title(f"{key}: Filtered ECG Lead II")
66           plt.xlabel('Time (s)')
67           plt.ylabel('Amplitude')
68           plt.show()
69   
70       return filtered_signal

73   # Function to calculate QT intervals.
74       # Process the ECG signal using neurokit2 (detects R-peaks, Q, and T points)
75       # ecg_process has many steps:
76       # 1) cleaning the ecg with ecg_clean()
77       # 2) peak detection with ecg_peaks()
78       # 3) HR calculus with signal_rate()
79       # 4) signal quality assessment with ecg_quality()
80       # 5) QRS elimination with ecg_delineate() and
81       # 6) cardiac phase determination with ecg_phase().
82   def calculateQtIntervals(key, filtered_signal, fs):
83       ecg_analysis, _ = nk.ecg_process(filtered_signal, sampling_rate=fs)
84       q_points = ecg_analysis['ECG_Q_Peaks'] # This is default output of the ecg_process.
85       t_points = ecg_analysis['ECG_T_Offsets'] # This is default output of the ecg_process.
86       q_indices = q_points[q_points == 1].index.to_list()
87       t_indices = t_points[t_points == 1].index.to_list()
88   
89       time = np.arange(filtered_signal.size) / fs
90   
91       # calculating QT intervals for all patients and saving them
92       qt_intervals = []
93       for q, t in zip(q_indices, t_indices):
94           if t > q:  # Ensure T point is after Q point for a valid QT interval
95               qt_interval = (t - q) / fs  # Convert sample indices to time
96               qt_intervals.append(qt_interval)
97   
98       # plotting filtered ecg and QT intervals for 3 patients only
99       if key in ["Patient1", "Patient2", "Patient3"]:
100          plt.figure(figsize=(10, 6)) # Determining a size of the figure
101          plt.plot(time, filtered_signal, label="Filtered ECG Lead II")
102          plt.title(f'{key} - Filtered ECG with QT Intervals')
103  
104          for q, t in zip(q_indices, t_indices):
105              if t > q:  # Ensure T point is after Q point for a valid QT interval
106  
107                  # Plot the QT interval as a red line segment
108                  plt.plot([q / fs, t / fs],
109                           [filtered_signal[q], filtered_signal[t]], color='red', lw=2,
110                           label='QT Interval' if len(qt_intervals) == 1 else "")  # Label only the first for legend clarity
111  
112          plt.xlabel('Time (s)')
113          plt.ylabel('Amplitude')
114          plt.legend()
115          plt.grid(True)
116          plt.tight_layout()
117          plt.show()
118          plt.close()
119  
120      return qt_intervals

123  # Function to calculate and store average QT interval
124  def calculateAverageQt(ecg_dict, fs_dict):
125      # The average QT intervals for all patients will be stored in average_qt_dict
126      average_qt_dict = {}
127      for key, ecg_signal in ecg_dict.items():
128          fs = fs_dict[key] # corresponding sampling freq. for each signal
129          filtered_signal = filterEcg(key, ecg_signal, fs) # This calls the filtering function
130          qt_intervals = calculateQtIntervals(key, filtered_signal, fs) # Calculates the intervals based on the filtered data
131          average_qt_interval = np.mean(qt_intervals) if qt_intervals else None # Calculates the average QT for each patient.
132          average_qt_dict[key] = average_qt_interval
133  
134      return average_qt_dict

137  # Function to load and reshape genotype data
138  # Reshaping to keep the structure for 7 rows
139  # New genotype starts every 7th row
140  def loadAndReshapeGenotype(filepath):
141      results = pd.read_csv(filepath, delimiter="\t", header=None)
142      selected = results.iloc[:, 1::7]  # Select every 7th column starting from index 1 --> each new genotype AA, AB or BB.
143      s_array = selected.values
144      reshaped = s_array.reshape(7, -1)  # Automatically calculate columns based on data size, when we want to have
145                                         # the original 7 rows.
146      return reshaped

149  # Function to extract QT intervals based on genotype AA, AB or BB
150  # Function goes through the reshaped data, consisting of 7 rows and 129 columns
151  # 129 is the number of different genotypes i.e. different patients
152  # One group to study = same genotype from one row. E.g. all BB genotypes from row 1.
153  def QtByGenotype(reshaped, average_qt_dict, row):
154      patients = list(average_qt_dict.keys())
155      genotypes = ["AB", "BB", "AA"] # making a list for looping
156      qt_results = {}
157  
158      for genotype in genotypes:
159          indices = np.where(reshaped[row, :] == genotype)[0]
160          qt_results[genotype] = [average_qt_dict[patients[idx]] for idx in indices
161                                  if average_qt_dict[patients[idx]] is not None]
162  
163      return qt_results["AB"], qt_results["BB"], qt_results["AA"]

166  def AnovaTest(qt_AB, qt_BB, qt_AA):
167  
168      if len(qt_AB) != 0 and len(qt_BB) != 0 and qt_AA != 0:
169          result = f_oneway(qt_AB, qt_BB, qt_AA)
170          print(f"Statistic: {result.statistic:.3f} and p-value: {result.pvalue:.3f}")
171      else:
172          print("Cannot be calculated :-(")
